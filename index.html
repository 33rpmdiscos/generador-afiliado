const app = firebase.initializeApp(firebaseConfig); const db = firebase.database(); const affiliateId = "33rpmdiscos-21"; // ID de afiliado de Amazon // Función para generar una URL de afiliado function generateAffiliateUrl() { const inputUrl = document.getElementById('productUrl').value; if (!inputUrl.includes("amazon.")) { alert("Por favor, introduce una URL válida de Amazon."); return; } // Agregar ID de afiliado const url = new URL(inputUrl); url.searchParams.set("tag", affiliateId); const userId = "demo-user"; // Este ID se obtiene con autenticación, aquí es estático para pruebas const generatedUrl = url.toString(); // Guardar en Firebase const urlId = db.ref(users/${userId}/urls).push().key; db.ref(users/${userId}/urls/${urlId}).set({ url: generatedUrl, compraValidada: false, ganancia: null }); document.getElementById('generatedUrl').innerHTML = `

URL generada: ${generatedUrl}

`; cargarHistorial(userId); } // Función para cargar el historial de URLs de un usuario function cargarHistorial(userId) { db.ref(users/${userId}/urls).once('value', snapshot => { const historyDiv = document.getElementById('history'); historyDiv.innerHTML = ''; snapshot.forEach(childSnapshot => { const data = childSnapshot.val(); const estado = data.compraValidada ? "Validada" : "Pendiente"; const ganancia = data.ganancia ? Ganancia: ${data.ganancia} € : "Ganancia: -"; const item = document.createElement('div'); item.innerHTML = `

${data.url}
Estado: ${estado}
${ganancia}

`; historyDiv.appendChild(item); }); }); } // Funciones de administración function cargarDatosAdmin() { db.ref('users').once('value', snapshot => { const adminTable = document.getElementById('adminTable'); adminTable.innerHTML = ''; snapshot.forEach(userSnapshot => { const userId = userSnapshot.key; const userUrls = userSnapshot.val().urls; for (const urlId in userUrls) { const data = userUrls[urlId]; const estado = data.compraValidada ? "Validada" : "Pendiente"; const ganancia = data.ganancia ? data.ganancia : 0; const row = document.createElement('tr'); row.innerHTML = ` ${userId} ${data.url} ${estado} ${ganancia} `; adminTable.appendChild(row); } }); }); } function validarCompra(userId, urlId) { db.ref(users/${userId}/urls/${urlId}).update({ compraValidada: true }) .then(() => alert('Compra validada')) .then(cargarDatosAdmin); } function asignarGanancia(userId, urlId) { const ganancia = prompt("Introduce la ganancia (€):"); if (ganancia && !isNaN(ganancia)) { db.ref(users/${userId}/urls/${urlId}).update({ ganancia: parseFloat(ganancia) }) .then(() => alert('Ganancia añadida')) .then(cargarDatosAdmin); } else { alert("Por favor, introduce un número válido."); } } // Cargar historial y datos de administración al inicio const userId = "demo-user"; // Este es un ID de prueba; se sustituye con autenticación cargarHistorial(userId); cargarDatosAdmin();
